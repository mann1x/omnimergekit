# Eval-only deps (loaded on top of requirements.txt)
# Includes lm-eval-harness with API extras for the `local-completions` and
# `local-chat-completions` model backends used in our recipes.
#
# Install:  pip install -r requirements-eval.txt
# Pin lm-eval to 0.4.11 — 0.4.12 needs Py3.13 (TypedDict.extra_items),
# 0.4.9 broke on transformers 5.x rename. See
# memory/feedback_pod_lmeval_py311.md.

lm-eval[api,math,ifeval]==0.4.11

# `math` extra installs sympy + math-verify + antlr4-python3-runtime==4.11
# required by minerva_math / math500 / aime templates. Without them the
# task class fails to import at lm-eval startup with ModuleNotFoundError
# even before any model is hit. Pin antlr4 explicitly because pip resolvers
# occasionally pull a newer incompatible runtime.
antlr4-python3-runtime==4.11.0
sympy>=1.13
math-verify>=0.9.0

# `ifeval` extra is supposed to pull these but pip resolution flakes on some
# manylinux wheels for nltk on Py3.11, leaving lm-eval to import-time fail on
# ifeval templates with `ModuleNotFoundError: langdetect`. Pin explicitly so
# every fresh pod gets a clean ifeval load.
# Caught on pod 36755693 (2026-05-14): reeval24k ifeval_reeval24k crashed at
# `from lm_eval.tasks.ifeval import instructions_registry` because langdetect
# was absent. ifeval template ran 0 samples; lm-eval still wrote a 0-row
# results.json. See memory/feedback_lm_eval_pod_deps.md.
langdetect>=1.0.9
immutabledict>=4.2.0
nltk>=3.9.1

# Code-eval support: lm_eval HE/MBPP need both HF_ALLOW_CODE_EVAL=1 AND
# --confirm_run_unsafe_code at invocation time. The package itself just
# needs the tenacity backoff helper and the optional human_eval scorer.
tenacity==9.1.4
human-eval>=1.0.3                # optional, only for the original HumanEval scorer

# Response cache. lm-eval 0.4.11 declares `sqlitedict` as a CORE (unconditional)
# dependency for its CachingLM `--use_cache` path, and our own eval/cache_sqlite.py,
# eval/lcb/lcb_llama_server.py resume store, and run_lcb100_mpe_chain.sh import it
# directly. A clean `pip install -r requirements-eval.txt` pulls it transitively,
# but a `--no-deps` / partial lm-eval install drops it silently — and EVERY omk_eval
# llama run passes `--use_cache`, so CachingLM then crashes at construction with
# `ModuleNotFoundError: sqlitedict`. Pin it explicitly so it survives, exactly like
# the langdetect/immutabledict/nltk re-pins above. Caught on the solidpc omk env
# 2026-05-30 (gsm8k_30 P5 verification). See memory/feedback_lm_eval_pod_deps.md.
sqlitedict>=2.1.0

# LiveCodeBench scorer (vendored runner is in eval/lcb/lcb_llama_server.py)
# pip install -e <livecodebench-repo> separately if you need the v3+ scorer

# Suite orchestrator (scripts/eval_suite_vllm.sh) needs PyYAML to compute
# per-template vLLM-config fingerprints. Inherited from requirements.txt
# (transformers pulls PyYAML transitively) — listed here for clarity.
pyyaml>=6.0
